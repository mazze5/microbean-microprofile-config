033/**
034 * An {@link AutoCloseable} implementation of the abstract {@link
035 * org.eclipse.microprofile.config.spi.ConfigProviderResolver} class.
036 *
037 * <h2>Thread Safety</h2>
038 *
039 * <p>This class is safe for concurrent use by multiple threads.</p>
040 *
041 * @author <a href="https://about.me/lairdnelson"
042 * target="_parent">Laird Nelson</a>
043 *
044 * @see org.eclipse.microprofile.config.spi.ConfigProviderResolver
045 */
046public final class ConfigProviderResolver extends org.eclipse.microprofile.config.spi.ConfigProviderResolver implements AutoCloseable {
047
048  // @GuardedBy("self")
049  private final Map<ClassLoader, Config> configMap;
050
051  /**
052   * Creates a new {@link ConfigProviderResolver}.
053   */
054  public ConfigProviderResolver() {
055    super();
056    this.configMap = new HashMap<>();
057  }
058
059  /**
060   * Closes this {@link ConfigProviderResolver} using a best-effort strategy.
061   *
062   * <p>This method attempts to {@linkplain #releaseConfig(Config)
063   * release} each of the {@link AutoCloseable} {@link Config}
064   * instances that are {@linkplain #registerConfig(Config,
065   * ClassLoader) registered} with it.  If any exception occurs, it is
066   * aggregated with any others and thrown at the end of the whole
067   * process.</p>
068   *
069   * <h2>Thread Safety</h2>
070   *
071   * <p>This method is idempotent and safe for concurrent use by
072   * multiple threads.</p>
073   *
074   * @exception RuntimeException if an error occurs
075   *
076   * @see #registerConfig(Config, ClassLoader)
077   *
078   * @see #releaseConfig(Config)
079   */
080  @Override
081  public final void close() {
082    synchronized (this.configMap) {
083      if (!this.configMap.isEmpty()) {
084        RuntimeException throwMe = null;
085        final Collection<Config> configs = new HashSet<>(this.configMap.values());
086        assert configs != null;
087        assert !configs.isEmpty();
088        for (final Config config : configs) {
089          try {
090            this.releaseConfig(config);
091          } catch (final RuntimeException runtimeException) {
092            if (throwMe == null) {
093              throwMe = runtimeException;
094            } else {
095              throwMe.addSuppressed(runtimeException);
096            }
097          } catch (final Exception exception) {
098            if (throwMe == null) {
099              throwMe = new RuntimeException(exception);
100            } else {
101              throwMe.addSuppressed(exception);
102            }
103          }
104        }
105        if (throwMe != null) {
106          throw throwMe;
107        }
108      }
109    }
110  }

112  /**
113   * Creates and returns a new {@link
114   * org.eclipse.microprofile.config.spi.ConfigBuilder}.
115   *
116   * <p>This method never returns {@code null}.</p>
117   *
118   * <h2>Thread Safety</h2>
119   *
120   * <p>This method is safe for concurrent use by multiple threads.</p>
121   *
122   * @return a new {@link
123   * org.eclipse.microprofile.config.spi.ConfigBuilder}; never {@code
124   * null}
125   */
126  @Override
127  public final org.eclipse.microprofile.config.spi.ConfigBuilder getBuilder() {
128    return new ConfigBuilder();
129  }
130
131  /**
132   * Returns the sole {@link Config} instance appropriate for the
133   * {@linkplain Thread#getContextClassLoader() context
134   * <code>ClassLoader</code>}, creating and building a default one
135   * just in time if necessary.
136   *
137   * <h2>Thread Safety</h2>
138   *
139   * <p>This method is safe for concurrent use by multiple threads.</p>
140   *
141   * <p>This method never returns {@code null}.</p>
142   *
143   * @return a {@link Config} instance; never {@code null}
144   *
145   * @see #getConfig(ClassLoader)
146   *
147   * @see Thread#getContextClassLoader()
148   *
149   * @see
150   * org.eclipse.microprofile.config.ConfigProvider#getConfig()
151   */
152  @Override
153  public final Config getConfig() {
154    return this.getConfig(AccessController.doPrivileged((PrivilegedAction<ClassLoader>)() -> Thread.currentThread().getContextClassLoader()));
155  }
156
157  /**
158   * Returns the sole {@link Config} instance appropriate for the
159   * supplied {@link ClassLoader}, creating and building a default one
160   * just in time if necessary.
161   *
162   * <h2>Thread Safety</h2>
163   *
164   * <p>This method is safe for concurrent use by multiple threads.</p>
165   *
166   * <h2>Implementation Notes</h2>
167   *
168   * <p>The specification does not indicate what to do if the supplied
169   * {@link ClassLoader} is {@code null}, but spread throughout other
170   * areas of the specification {@linkplain ConfigProvider#getConfig()
171   * there are implications} that a {@code null} {@link ClassLoader}
172   * means that the current thread's {@linkplain
173   * Thread#getContextClassLoader() context classloader} should be
174   * used instead.  This implementation follows those implications.</p>
175   *
176   * @param classLoader the {@link ClassLoader} used to identify the
177   * {@link Config} to return; may be {@code null} in which case the
178   * {@linkplain Thread#getContextClassLoader() context
179   * <code>ClassLoader</code>} will be used instead
180   *
181   * @return a {@link Config} instance; never {@code null}
182   *
183   * @see #getBuilder()
184   *
185   * @see #getConfig()
186   */
187  @Override
188  public final Config getConfig(ClassLoader classLoader) {
189    if (classLoader == null) {
190      classLoader = AccessController.doPrivileged((PrivilegedAction<ClassLoader>)() -> Thread.currentThread().getContextClassLoader());
191    }
192    Config returnValue = null;
193    synchronized (this.configMap) {
194      returnValue = configMap.get(classLoader);
195      if (returnValue == null) {
196        returnValue = this.getBuilder()
197          .addDefaultSources()
198          .addDiscoveredSources()
199          .addDiscoveredConverters()
200          .forClassLoader(classLoader)
201          .build();
202        assert returnValue != null;
203        // Deliberately called with the lock held on the configMap
204        this.registerConfig(returnValue, classLoader);
205      }
206    }
207    return returnValue;
208  }

210  /**
211   * Registers the supplied {@link Config} instance under the supplied
212   * {@link ClassLoader} in some way.
213   *
214   * <h2>Thread Safety</h2>
215   *
216   * <p>This method is safe for concurrent use by multiple threads.</p>
217   *
218   * <h2>Implementation Notes</h2>
219   *
220   * <p>The specification says:</p>
221   *
222   * <blockquote>If the ClassLoader is null then the current
223   * Application will be used.</blockquote>
224   *
225   * <p>This implementation assumes "current Application" is
226   * equivalent to "current thread's {@linkplain
227   * Thread#getContextClassLoader() context
228   * <code>ClassLoader</code>}".</p>
229   *
230   * @param config the {@link Config} to register; may be {@code null}
231   * in which case no action will be taken
232   *
233   * @param classLoader the {@link ClassLoader} to use as the key
234   * under which the supplied {@link Config} should be registered; if
235   * {@code null} then the return value of {@link
236   * Thread#getContextClassLoader()} will be used instead
237   *
238   * @see
239   * org.eclipse.microprofile.config.spi.ConfigProviderResolver#registerConfig(Config,
240   * ClassLoader)
241   */
242  @Override
243  public final void registerConfig(final Config config, ClassLoader classLoader) {
244    if (config != null) {
245      if (classLoader == null) {
246        classLoader = AccessController.doPrivileged((PrivilegedAction<ClassLoader>)() -> Thread.currentThread().getContextClassLoader());
247      }
248      synchronized (this.configMap) {
249        final Config oldConfig = this.configMap.putIfAbsent(classLoader, config);
250        if (oldConfig != null) {          
251          // The specification says that an IllegalStateException
252          // should be thrown "if there is already a Config registered
253          // within the Application."  It is not entirely clear what
254          // "within the Application" means.
255          throw new IllegalStateException("configMap.containsKey(" + classLoader + "): " + oldConfig);
256        }
257      }
258    }
259  }

261  /**
262   * Releases all {@link Config} instances associated in any way with
263   * the supplied {@link ClassLoader}.
264   *
265   * <h2>Thread Safety</h2>
266   *
267   * <p>This method is safe for concurrent use by multiple threads.</p>
268   *
269   * <h2>Implementation Notes</h2>
270   *
271   * <p>The specification says:</p>
272   *
273   * <blockquote>A Config normally gets released if the Application it
274   * is associated with gets destroyed.</blockquote>
275   *
276   * <p>This implementation assumes the previous sentence means that
277   * if a {@link Config} was previously {@linkplain
278   * #registerConfig(Config, ClassLoader) registered} under a {@link
279   * ClassLoader} <em>A</em>, and under a {@link ClassLoader}
280   * <em>B</em>, then both registrations will be deleted.  The
281   * equivalence between "Application" and {@link ClassLoader} is
282   * derived from the documentation for the {@link
283   * ConfigProvider#getConfig()} method, where there is an implication
284   * that the {@linkplain Thread#getContextClassLoader() context
285   * classloader} will be used as the key under which the {@link
286   * Config} to be returned may be found.</p>
287   *
288   * <p>{@linkplain ConfigProvider#getConfig(ClassLoader) Elsewhere in
289   * the specification}, it is stated:</p>
290   *
291   * <blockquote>There is exactly a single Config instance per
292   * ClassLoader[.]</blockquote>
293   *
294   * <p>This is of course false since there may be zero {@link Config}
295   * instances per {@link ClassLoader}.  Leaving that aside, this
296   * sentence also does not say whether it is permitted for a single
297   * {@link Config} instance to be shared between two {@link
298   * ClassLoader}s.  This implementation permits such perhaps edge
299   * cases, but a call to this {@link #releaseConfig(Config)} method
300   * will remove both such registrations if present.</p>
301   *
302   * <p>The specification does not indicate whether the supplied
303   * {@link Config} must be non-{@code null}.  This implementation
304   * consequently accepts {@code null} {@link Config}s and does
305   * nothing in such cases.</p>
306   *
307   * <p>This method is called by the {@link #close()} method.
308   *
309   * @param config the {@link Config} to release; may be {@code null}
310   * in which case no action will be taken
311   *
312   * @see #close()
313   *
314   * @see #registerConfig(Config, ClassLoader)
315   */
316  @Override
317  public final void releaseConfig(final Config config) {
318    // The specification says nothing about whether arguments can be null.
319    if (config != null) {
320      RuntimeException throwMe = null;
321      synchronized (this.configMap) {
322        if (!this.configMap.isEmpty()) {
323          final Set<? extends Entry<?, ? extends Config>> entrySet = this.configMap.entrySet();
324          assert entrySet != null;
325          assert !entrySet.isEmpty();
326          final Iterator<? extends Entry<?, ? extends Config>> entryIterator = entrySet.iterator();
327          assert entryIterator != null;
328          while (entryIterator.hasNext()) {
329            final Entry<?, ? extends Config> entry = entryIterator.next();
330            if (entry != null) {
331              final Config existingConfig = entry.getValue();
332              if (config.equals(existingConfig)) {
333                entryIterator.remove();
334                if (existingConfig instanceof AutoCloseable) {
335                  try {
336                    ((AutoCloseable)existingConfig).close();
337                  } catch (final RuntimeException runtimeException) {
338                    if (throwMe == null) {
339                      throwMe = runtimeException;
340                    } else {
341                      throwMe.addSuppressed(runtimeException);
342                    }
343                  } catch (final Exception exception) {
344                    if (throwMe == null) {
345                      throwMe = new RuntimeException(exception.getMessage(), exception);
346                    } else {
347                      throwMe.addSuppressed(exception);
348                    }
349                  }
350                }
351              }
352            }
353          }
354        }
355      }
356      if (throwMe != null) {
357        throw throwMe;
358      }
359    }
360  }
